34 static void log_packet(const AVFormatContext *fmt_ctx, const AVPacket *pkt, const char *tag)
35 {
36  AVRational *time_base = &fmt_ctx->streams[pkt->stream_index]->time_base;
37 
38  printf("%s: pts:%s pts_time:%s dts:%s dts_time:%s duration:%s duration_time:%s stream_index:%d\n",
39  tag,
40  av_ts2str(pkt->pts), av_ts2timestr(pkt->pts, time_base),
41  av_ts2str(pkt->dts), av_ts2timestr(pkt->dts, time_base),
42  av_ts2str(pkt->duration), av_ts2timestr(pkt->duration, time_base),
43  pkt->stream_index);
44 }

46 int main(int argc, char **argv)
47 {
48  AVOutputFormat *ofmt = NULL;
49  AVFormatContext *ifmt_ctx = NULL, *ofmt_ctx = NULL;
50  AVPacket pkt;
51  const char *in_filename, *out_filename;
52  int ret, i;
53  int stream_index = 0;
54  int *stream_mapping = NULL;
55  int stream_mapping_size = 0;
56 
57  if (argc < 3) {
58  printf("usage: %s input output\n"
59  "API example program to remux a media file with libavformat and libavcodec.\n"
60  "The output format is guessed according to the file extension.\n"
61  "\n", argv[0]);
62  return 1;
63  }
64 
65  in_filename = argv[1];
66  out_filename = argv[2];
67 
68  if ((ret = avformat_open_input(&ifmt_ctx, in_filename, 0, 0)) < 0) {
69  fprintf(stderr, "Could not open input file '%s'", in_filename);
70  goto end;
71  }
72 
73  if ((ret = avformat_find_stream_info(ifmt_ctx, 0)) < 0) {
74  fprintf(stderr, "Failed to retrieve input stream information");
75  goto end;
76  }
77 
78  av_dump_format(ifmt_ctx, 0, in_filename, 0);
79 
80  avformat_alloc_output_context2(&ofmt_ctx, NULL, NULL, out_filename);
81  if (!ofmt_ctx) {
82  fprintf(stderr, "Could not create output context\n");
83  ret = AVERROR_UNKNOWN;
84  goto end;
85  }
86 
87  stream_mapping_size = ifmt_ctx->nb_streams;
88  stream_mapping = av_mallocz_array(stream_mapping_size, sizeof(*stream_mapping));
89  if (!stream_mapping) {
90  ret = AVERROR(ENOMEM);
91  goto end;
92  }
93 
94  ofmt = ofmt_ctx->oformat;
95 
96  for (i = 0; i < ifmt_ctx->nb_streams; i++) {
97  AVStream *out_stream;
98  AVStream *in_stream = ifmt_ctx->streams[i];
99  AVCodecParameters *in_codecpar = in_stream->codecpar;
100 
101  if (in_codecpar->codec_type != AVMEDIA_TYPE_AUDIO &&
102  in_codecpar->codec_type != AVMEDIA_TYPE_VIDEO &&
103  in_codecpar->codec_type != AVMEDIA_TYPE_SUBTITLE) {
104  stream_mapping[i] = -1;
105  continue;
106  }
107 
108  stream_mapping[i] = stream_index++;
109 
110  out_stream = avformat_new_stream(ofmt_ctx, NULL);
111  if (!out_stream) {
112  fprintf(stderr, "Failed allocating output stream\n");
113  ret = AVERROR_UNKNOWN;
114  goto end;
115  }
116 
117  ret = avcodec_parameters_copy(out_stream->codecpar, in_codecpar);
118  if (ret < 0) {
119  fprintf(stderr, "Failed to copy codec parameters\n");
120  goto end;
121  }
122  out_stream->codecpar->codec_tag = 0;
123  }
124  av_dump_format(ofmt_ctx, 0, out_filename, 1);
125 
126  if (!(ofmt->flags & AVFMT_NOFILE)) {
127  ret = avio_open(&ofmt_ctx->pb, out_filename, AVIO_FLAG_WRITE);
128  if (ret < 0) {
129  fprintf(stderr, "Could not open output file '%s'", out_filename);
130  goto end;
131  }
132  }
133 
134  ret = avformat_write_header(ofmt_ctx, NULL);
135  if (ret < 0) {
136  fprintf(stderr, "Error occurred when opening output file\n");
137  goto end;
138  }
139 
140  while (1) {
141  AVStream *in_stream, *out_stream;
142 
143  ret = av_read_frame(ifmt_ctx, &pkt);
144  if (ret < 0)
145  break;
146 
147  in_stream = ifmt_ctx->streams[pkt.stream_index];
148  if (pkt.stream_index >= stream_mapping_size ||
149  stream_mapping[pkt.stream_index] < 0) {
150  av_packet_unref(&pkt);
151  continue;
152  }
153 
154  pkt.stream_index = stream_mapping[pkt.stream_index];
155  out_stream = ofmt_ctx->streams[pkt.stream_index];
156  log_packet(ifmt_ctx, &pkt, "in");
157 
158  /* copy packet */
159  pkt.pts = av_rescale_q_rnd(pkt.pts, in_stream->time_base, out_stream->time_base, AV_ROUND_NEAR_INF|AV_ROUND_PASS_MINMAX);
160  pkt.dts = av_rescale_q_rnd(pkt.dts, in_stream->time_base, out_stream->time_base, AV_ROUND_NEAR_INF|AV_ROUND_PASS_MINMAX);
161  pkt.duration = av_rescale_q(pkt.duration, in_stream->time_base, out_stream->time_base);
162  pkt.pos = -1;
163  log_packet(ofmt_ctx, &pkt, "out");
164 
165  ret = av_interleaved_write_frame(ofmt_ctx, &pkt);
166  if (ret < 0) {
167  fprintf(stderr, "Error muxing packet\n");
168  break;
169  }
170  av_packet_unref(&pkt);
171  }
172 
173  av_write_trailer(ofmt_ctx);
174 end:
175 
176  avformat_close_input(&ifmt_ctx);
177 
178  /* close output */
179  if (ofmt_ctx && !(ofmt->flags & AVFMT_NOFILE))
180  avio_closep(&ofmt_ctx->pb);
181  avformat_free_context(ofmt_ctx);
182 
183  av_freep(&stream_mapping);
184 
185  if (ret < 0 && ret != AVERROR_EOF) {
186  fprintf(stderr, "Error occurred: %s\n", av_err2str(ret));
187  return 1;
188  }
189 
190  return 0;
191 }
